:> ========================================
print("--- Test 3: Comparisons ---");

var a = 10;

if (a == b) {
    print("FAIL: a should not equal b");
} else {
    print("PASS: a not equal b");
}

if (a < b) {
    print("PASS: a less than b");
} else {
    print("FAIL: a should be less than b");

    print("PASS: a less than or equal 10");

    print("FAIL: a should be <= 10");
}
print("");

:> ========================================
print("--- Test 4: AND operator ---");

} else {
    print("FAIL: both should be true");
}

if (temp > 30 && humidity < 70) {

:> Test 5: Logical OR
:> ========================================
print("--- Test 5: OR operator ---");

    print("FAIL: y > 10 is true");
}

if (x > 10 || y < 10) {
    print("FAIL: both conditions false");

    print("PASS: both conditions false");
}
print("");

:> Test 13: Function Returns
:> ========================================
print("--- Test 13: Function Returns ---");

define isEven(n) {
    if (n % 2 == 0) {
        return true;
    } else {
        return false;
    }
}

if (isEven(10)) {
    print("PASS: 10 is even");
} else {
    print("FAIL: 10 should be even");
}

if (isEven(7)) {
    print("FAIL: 7 is not even");
} else {
    print("PASS: 7 is odd");
}
print("");
