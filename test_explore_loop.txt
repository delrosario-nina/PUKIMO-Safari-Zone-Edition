// Setup Safari Zone
var zone = SafariZone(balls=5, turns=10);  // 5 balls, 10 turns
zone.pokemon->add("Pikachu");
zone.pokemon->add("Charizard");
zone.pokemon->add("Bulbasaur");
zone.pokemon->add("Squirtle");
zone.pokemon->add("Eevee");

print ("Welcome to the Safari Zone!");
print ("You have " + zone.balls + " Safari Balls");

var caughtPokemon = Team("Player");

explore() {
    print ("");
    print( "--- Exploring Safari Zone ---");
    print ("Balls left: " + zone.balls);
    print ("Pokemon in zone: " + zone.pokemon->list());

    // Check if we should leave
    if (zone.balls <= 0) {
        print ("Out of balls! Leaving Safari Zone.");
        run;
    }

    // Try to catch a Pokemon
    var pokemon = throwBall(zone);
    caughtPokemon.pokemon->add(pokemon);

    // Use resources
    zone->useBall();
    zone->useTurn();

    print ("Caught so far: " + caughtPokemon.pokemon->list());
}

print ("");
print ("=== Safari Zone Session Complete ===");
print ("Pokemon caught: " + caughtPokemon.pokemon->list());
print ("Total: " + caughtPokemon.pokemon->count() + " Pokemon");
